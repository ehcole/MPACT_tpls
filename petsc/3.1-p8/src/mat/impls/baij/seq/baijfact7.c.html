257:   PetscMemzero(rtmp,bs2*n*sizeof(MatScalar));
258:   ics  = ic;

260:   for (i=0; i<n; i++){
261:     /* zero rtmp */

266:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));

544:       pv[24] = x[24]; pv[25] = x[25]; pv[26] = x[26]; pv[27] = x[27];
545:       pv[28] = x[28]; pv[29] = x[29]; pv[30] = x[30]; pv[31] = x[31];
546:       pv[32] = x[32]; pv[33] = x[33]; pv[34] = x[34]; pv[35] = x[35];

548:     }

564: PetscErrorCode MatLUFactorNumeric_SeqBAIJ_6_NaturalOrdering(Mat B,Mat A,const MatFactorInfo *info)
565: {
566:   Mat            C=B;
567:   Mat_SeqBAIJ    *a=(Mat_SeqBAIJ*)A->data,*b=(Mat_SeqBAIJ *)C->data;
569:   PetscInt       i,j,k,nz,nzL,row;
570:   const PetscInt n=a->mbs,*ai=a->i,*aj=a->j,*bi=b->i,*bj=b->j;
571:   const PetscInt *ajtmp,*bjtmp,*bdiag=b->diag,*pj,bs2=a->bs2;
572:   MatScalar      *rtmp,*pc,*mwork,*v,*pv,*aa=a->a;
573:   PetscInt       flg;
574:   PetscReal      shift = info->shiftamount;

577:   /* generate work space needed by the factorization */
578:   PetscMalloc2(bs2*n,MatScalar,&rtmp,bs2,MatScalar,&mwork);
579:   PetscMemzero(rtmp,bs2*n*sizeof(MatScalar));

581:   for (i=0; i<n; i++){
582:     /* zero rtmp */
583:     /* L part */
584:     nz    = bi[i+1] - bi[i];
585:     bjtmp = bj + bi[i];
586:     for  (j=0; j<nz; j++){
587:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
588:     }

590:     /* U part */
591:     nz = bdiag[i] - bdiag[i+1];
592:     bjtmp = bj + bdiag[i+1]+1;
593:     for  (j=0; j<nz; j++){
594:       PetscMemzero(rtmp+bs2*bjtmp[j],bs2*sizeof(MatScalar));
595:     }
596: 
597:     /* load in initial (unfactored row) */
598:     nz    = ai[i+1] - ai[i];
599:     ajtmp = aj + ai[i];
600:     v     = aa + bs2*ai[i];
601:     for (j=0; j<nz; j++) {
602:       PetscMemcpy(rtmp+bs2*ajtmp[j],v+bs2*j,bs2*sizeof(MatScalar));
603:     }

605:     /* elimination */
606:     bjtmp = bj + bi[i];
607:     nzL   = bi[i+1] - bi[i];
608:     for(k=0;k < nzL;k++) {
609:       row = bjtmp[k];
610:       pc = rtmp + bs2*row;
611:       for (flg=0,j=0; j<bs2; j++) { if (pc[j]!=0.0) { flg = 1; break; }}
612:       if (flg) {
613:         pv = b->a + bs2*bdiag[row];
614:         /* Kernel_A_gets_A_times_B(bs,pc,pv,mwork); *pc = *pc * (*pv); */
615:         Kernel_A_gets_A_times_B_6(pc,pv,mwork);
616: 
617:         pj = b->j + bdiag[row+1]+1; /* begining of U(row,:) */
618:         pv = b->a + bs2*(bdiag[row+1]+1);
619:         nz = bdiag[row] - bdiag[row+1] - 1; /* num of entries inU(row,:), excluding diag */
620:         for (j=0; j<nz; j++) {
621:           /* Kernel_A_gets_A_minus_B_times_C(bs,rtmp+bs2*pj[j],pc,pv+bs2*j); */
622:           /* rtmp+bs2*pj[j] = rtmp+bs2*pj[j] - (*pc)*(pv+bs2*j) */
623:           v    = rtmp + bs2*pj[j];
624:           Kernel_A_gets_A_minus_B_times_C_6(v,pc,pv);
625:           pv  += bs2;
626:         }
627:         PetscLogFlops(432*nz+396); /* flops = 2*bs^3*nz + 2*bs^3 - bs2) */
628:       }
629:     }

631:     /* finished row so stick it into b->a */
632:     /* L part */
633:     pv   = b->a + bs2*bi[i] ;
634:     pj   = b->j + bi[i] ;
635:     nz   = bi[i+1] - bi[i];
636:     for (j=0; j<nz; j++) {
637:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
638:     }

640:     /* Mark diagonal and invert diagonal for simplier triangular solves */
641:     pv   = b->a + bs2*bdiag[i];
642:     pj   = b->j + bdiag[i];
643:     PetscMemcpy(pv,rtmp+bs2*pj[0],bs2*sizeof(MatScalar));
644:     /* Kernel_A_gets_inverse_A(bs,pv,v_pivots,v_work); */
645:     Kernel_A_gets_inverse_A_6(pv,shift);
646: 
647:     /* U part */
648:     pv = b->a + bs2*(bdiag[i+1]+1);
649:     pj = b->j + bdiag[i+1]+1;
650:     nz = bdiag[i] - bdiag[i+1] - 1;
651:     for (j=0; j<nz; j++){
652:       PetscMemcpy(pv+bs2*j,rtmp+bs2*pj[j],bs2*sizeof(MatScalar));
653:     }
654:   }
655:   PetscFree2(rtmp,mwork);
656:   C->ops->solve          = MatSolve_SeqBAIJ_6_NaturalOrdering;
657:   C->ops->solvetranspose = MatSolveTranspose_SeqBAIJ_6_NaturalOrdering;
658:   C->assembled = PETSC_TRUE;
659:   PetscLogFlops(1.333333333333*6*6*6*n); /* from inverting diagonal blocks */
660:   return(0);
661: }
