  2: static char help[] = "Tests MatTranspose()\n\n";

  8: int main(int argc,char **args)
  9: {
 10:   Mat            C,A;
 11:   PetscScalar    v;
 12:   PetscInt       i,j,m = 4,n = 4,Ii,J,Istart,Iend;
 13:   PetscMPIInt    rank,size;
 15:   PetscTruth     equal=PETSC_FALSE;

 17:   PetscInitialize(&argc,&args,(char *)0,help);
 18:   MPI_Comm_rank(PETSC_COMM_WORLD,&rank);
 19:   MPI_Comm_size(PETSC_COMM_WORLD,&size);

 21:   PetscOptionsGetInt(PETSC_NULL,"-m",&m,PETSC_NULL);
 22:   n    = m;
 23: 
 24:   MatCreateMPIAIJ(PETSC_COMM_WORLD,PETSC_DECIDE,PETSC_DECIDE,
 25:          m*n,m*n,5,PETSC_NULL,5,PETSC_NULL,&C);

 27:   /* create the symmetric matrix for the five point stencil */
 28:   MatGetOwnershipRange(C,&Istart,&Iend);
 29:   for (Ii=Istart; Ii<Iend; Ii++) {
 30:     v = -1.0; i = Ii/n; j = Ii - i*n;
 31:     if (i>0)   {J = Ii - n; MatSetValues(C,1,&Ii,1,&J,&v,INSERT_VALUES);}
 32:     if (i<m-1) {J = Ii + n; MatSetValues(C,1,&Ii,1,&J,&v,INSERT_VALUES);}
 33:     if (j>0)   {J = Ii - 1; MatSetValues(C,1,&Ii,1,&J,&v,INSERT_VALUES);}
 34:     if (j<n-1) {J = Ii + 1; MatSetValues(C,1,&Ii,1,&J,&v,INSERT_VALUES);}
 35:     v = 4.0; MatSetValues(C,1,&Ii,1,&Ii,&v,INSERT_VALUES);
 36:   }
 37:   MatAssemblyBegin(C,MAT_FINAL_ASSEMBLY);
 38:   MatAssemblyEnd(C,MAT_FINAL_ASSEMBLY);
 39: 
 40:   MatTranspose(C,MAT_INITIAL_MATRIX,&A);

 42:   MatEqual(C,A,&equal);
 43:   if (!equal) SETERRQ(PETSC_ERR_SUP,"C != C^T");

 45:   MatDestroy(C);
 46:   MatDestroy(A);
 47:   PetscFinalize();
 48:   return 0;
 49: }
