runex1:
        -@${MPIEXEC} -n 1  ./ex1  > ex1.tmp 2>&1;\
           if (${DIFF} output/ex1.out ex1.tmp) then true; \
           else echo "Possible problem with ex1, diffs above"; fi ;\
           ${RM} -f ex1.tmp
