  1: !  Program usage: mpirun -np 1 chwirut1f [-help] [all TAO options]
  2: !
  3: !  Description:  This example demonstrates use of the TAO package to solve a
  4: !  nonlinear least-squares problem on a single processor.  We minimize the
  5: !  Chwirut function:
  6: !       sum_{i=0}^{n/2-1} ( alpha*(x_{2i+1}-x_{2i}^2)^2 + (1-x_{2i})^2 )
  7: !
  8: !  The C version of this code is chwirut1.c
  9: !
 10: !/*T
 11: !  Concepts: TAO^Solving an unconstrained minimization problem
 12: !  Routines: TaoCreate();
 13: !  Routines: TaoSetType();
 14: !  Routines: TaoSetSeparableObjectiveRoutine();
 15: !  Routines: TaoSetInitialVector();
 16: !  Routines: TaoSetFromOptions();
 17: !  Routines: TaoSolve();
 18: !  Routines: TaoDestroy();
 19: !  Processors: n
 20: !T*/
 21: !
 22: ! ----------------------------------------------------------------------
 23: !
 24:       implicit none

 26: #include "chwirut2f.h"

 28: ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
 29: !                   Variable declarations
 30: ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
 31: !
 32: !  See additional variable declarations in the file chwirut2f.h

 34:       PetscErrorCode   ierr    ! used to check for functions returning nonzeros
 35:       Vec              x       ! solution vector
 36:       Vec              f       ! vector of functions
 37:       Tao        tao     ! Tao context
 38:       PetscInt         i




 43: !  Note: Any user-defined Fortran routines (such as FormGradient)
 44: !  MUST be declared as external.

 46:       external FormFunction

 48: !  Initialize TAO and PETSc
 49:       call PetscInitialize(PETSC_NULL_CHARACTER,ierr)

 51:       call MPI_Comm_size(PETSC_COMM_WORLD,size,ierr)
 52:       CHKERRQ(ierr)
 53:       call MPI_Comm_rank(PETSC_COMM_WORLD,rank,ierr)
 54:       CHKERRQ(ierr)

 56: !  Initialize problem parameters
 57:       call InitializeData()

 59:       if (rank .eq. 0) then
 60: !  Allocate vectors for the solution and gradient
 61:          call VecCreateSeq(PETSC_COMM_SELF,n,x,ierr)
 62:          CHKERRQ(ierr)
 63:          call VecCreateSeq(PETSC_COMM_SELF,m,f,ierr)
 64:          CHKERRQ(ierr)


 67: !     The TAO code begins here

 69: !     Create TAO solver
 70:          call TaoCreate(PETSC_COMM_SELF,tao,ierr)
 71:          CHKERRQ(ierr)
 72:          call TaoSetType(tao,TAOPOUNDERS,ierr)
 73:          CHKERRQ(ierr)

 75: !     Set routines for function, gradient, and hessian evaluation
 76:          call TaoSetSeparableObjectiveRoutine(tao,f,                    &
 77:      &        FormFunction,PETSC_NULL_OBJECT,ierr)
 78:          CHKERRQ(ierr)

 80: !     Optional: Set initial guess
 81:          call FormStartingPoint(x)
 82:          call TaoSetInitialVector(tao, x, ierr)
 83:          CHKERRQ(ierr)


 86: !     Check for TAO command line options
 87:          call TaoSetFromOptions(tao,ierr)
 88:          CHKERRQ(ierr)
 89: !     SOLVE THE APPLICATION
 90:          call TaoSolve(tao,ierr)
 91:          CHKERRQ(ierr)

 93: !     Free TAO data structures
 94:          call TaoDestroy(tao,ierr)
 95:          CHKERRQ(ierr)

 97: !     Free PETSc data structures
 98:          call VecDestroy(x,ierr)
 99:          CHKERRQ(ierr)
100:          call VecDestroy(f,ierr)
101:          CHKERRQ(ierr)
102:          call StopWorkers(ierr)
103:          CHKERRQ(ierr)

105:       else
106:          call TaskWorker(ierr)
107:          CHKERRQ(ierr)
108:       endif

110:       call PetscFinalize(ierr)
111:       end

115: !  FormFunction - Evaluates the function f(X) and gradient G(X)
116: !
117: !  Input Parameters:
118: !  tao - the Tao context

120: !  dummy - not used

122: !  Output Parameters:

125:       subroutine FormFunction(tao, x, f, dummy, ierr)
126:       implicit none

128: ! n,alpha defined in chwirut2f.h
129: #include "chwirut2f.h"

131:       Tao        tao
132:       Vec              x,f
133:       PetscErrorCode   ierr
134:       PetscInt         dummy

136:       PetscInt         i,checkedin
137:       PetscInt         finished_tasks
138:       integer          next_task,status(MPI_STATUS_SIZE),tag,source

140: ! PETSc's VecGetArray acts differently in Fortran than it does in C.
141: ! Calling VecGetArray((Vec) X, (PetscReal) x_array(0:1), (PetscOffset) x_index, ierr)
142: ! will return an array of doubles referenced by x_array offset by x_index.
143: !  i.e.,  to reference the kth element of X, use x_array(k + x_index).
144: ! Notice that by declaring the arrays with range (0:1), we are using the C 0-indexing practice.
145:       PetscReal        f_v(0:1),x_v(0:1),fval
146:       PetscOffset      f_i,x_i

148:       0

150: !     Get pointers to vector data
151:       call VecGetArray(x,x_v,x_i,ierr)
152:       CHKERRQ(ierr)
153:       call VecGetArray(f,f_v,f_i,ierr)
154:       CHKERRQ(ierr)


157: !     Compute F(X)
158:       if (size .eq. 1) then
159:          ! Single processor
160:          do i=0,m-1
161:             call RunSimulation(x_v(x_i),i,f_v(i+f_i),ierr)
162:          enddo
163:       else
164:          ! Multiprocessor master
165:          next_task = 0
166:          finished_tasks = 0
167:          checkedin = 0

169:          do while (finished_tasks .lt. m .or. checkedin .lt. size-1)
170:             call MPI_Recv(fval,1,MPIU_SCALAR,MPI_ANY_SOURCE,               &
171:      &           MPI_ANY_TAG,PETSC_COMM_WORLD,status,ierr)
172:             tag = status(MPI_TAG)
173:             source = status(MPI_SOURCE)
174:             if (tag .eq. IDLE_TAG) then
175:                checkedin = checkedin + 1
176:             else
177:                f_v(f_i+tag) = fval
178:                finished_tasks = finished_tasks + 1
179:             endif
180:             if (next_task .lt. m) then
181:                ! Send task to worker
182:                call MPI_Send(x_v(x_i),n,MPIU_SCALAR,source,next_task,             &
183:      &              PETSC_COMM_WORLD,ierr)
184:                next_task = next_task + 1
185:             else
186:                ! Send idle message to worker
187:                call MPI_Send(x_v(x_i),n,MPIU_SCALAR,source,IDLE_TAG,              &
188:      &              PETSC_COMM_WORLD,ierr)
189:             end if
190:          enddo
191:       endif

193: !     Restore vectors
194:       call VecRestoreArray(x,x_v,x_i,ierr)
195:       CHKERRQ(ierr)
196:       call VecRestoreArray(F,f_v,f_i,ierr)
197:       CHKERRQ(ierr)
198:       return
199:       end

453:       subroutine TaskWorker(ierr)
454:       implicit none
455: #include "chwirut2f.h"
456:       PetscErrorCode ierr
457:       PetscReal x(n),f
458:       integer tag
459:       PetscInt index
460:       integer status(MPI_STATUS_SIZE)

462:       tag = IDLE_TAG
463:       f   = 0.0
464:       ! Send check-in message to master
465:       call MPI_Send(f,1,MPIU_SCALAR,0,IDLE_TAG,PETSC_COMM_WORLD,ierr)
466:       CHKERRQ(ierr)
467:       do while (tag .ne. DIE_TAG)
468:          call MPI_Recv(x,n,MPIU_SCALAR,0,MPI_ANY_TAG,PETSC_COMM_WORLD,     &
469:      &        status,ierr)
470:          CHKERRQ(ierr)
471:          tag = status(MPI_TAG)
472:          if (tag .eq. IDLE_TAG) then
473:             call MPI_Send(f,1,MPIU_SCALAR,0,IDLE_TAG,PETSC_COMM_WORLD,     &
474:      &           ierr)
475:             CHKERRQ(ierr)
476:          else if (tag .ne. DIE_TAG) then
477:             index = tag
478:             ! Compute local part of residual
479:             call RunSimulation(x,index,f,ierr)
480:             CHKERRQ(ierr)

482:             ! Return residual to master
483:             call MPI_Send(f,1,MPIU_SCALAR,0,tag,PETSC_COMM_WORLD,ierr)
484:             CHKERRQ(ierr)
485:          end if
486:       enddo
487:       0
488:       return
489:       end



493:       subroutine RunSimulation(x,i,f,ierr)
494:       implicit none
495: #include "chwirut2f.h"
496:       PetscReal x(n),f
497:       PetscInt i
498:       PetscErrorCode ierr
499:       f = y(i) - exp(-x(1)*t(i))/(x(2)+x(3)*t(i))
500:       0
501:       return
502:       end

504:       subroutine StopWorkers(ierr)
505:       implicit none
506: #include "chwirut2f.h"
507:       integer checkedin
508:       integer status(MPI_STATUS_SIZE)
509:       integer source
510:       PetscReal f,x(n)
511:       PetscErrorCode ierr
512:       PetscInt i

514:       checkedin=0
515:       do while (checkedin .lt. size-1)
516:          call MPI_Recv(f,1,MPIU_SCALAR,MPI_ANY_SOURCE,MPI_ANY_TAG,         &
517:      &        PETSC_COMM_WORLD,status,ierr)
518:          CHKERRQ(ierr)
519:          checkedin=checkedin+1
520:          source = status(MPI_SOURCE)
521:          do i=1,n
522:            x(i) = 0.0
523:          enddo
524:          call MPI_Send(x,n,MPIU_SCALAR,source,DIE_TAG,PETSC_COMM_WORLD,    &
525:      &        ierr)
526:          CHKERRQ(ierr)
527:       enddo
528:       ierr=0
529:       return
530:       end
