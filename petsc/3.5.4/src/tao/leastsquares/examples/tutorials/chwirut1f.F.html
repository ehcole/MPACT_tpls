  1: !  Program usage: mpirun -np 1 chwirut1f [-help] [all TAO options]
  2: !
  3: !  Description:  This example demonstrates use of the TAO package to solve a
  4: !  nonlinear least-squares problem on a single processor.  We minimize the
  5: !  Chwirut function:
  6: !       sum_{i=0}^{n/2-1} ( alpha*(x_{2i+1}-x_{2i}^2)^2 + (1-x_{2i})^2 )
  7: !
  8: !  The C version of this code is test_chwirut1.c

 11: !  Concepts: TAO^Solving an unconstrained minimization problem
 12: !  Routines: TaoCreate();

 20: !  Processors: 1

 52: !  MUST be declared as external.

 54:       external FormFunction

 58: !  Initialize TAO and PETSc
 59:       call PetscInitialize(PETSC_NULL_CHARACTER,ierr)

 61:       call MPI_Comm_size(PETSC_COMM_WORLD,size,ierr)
 62:       call MPI_Comm_rank(PETSC_COMM_WORLD,rank,ierr)
 63:       if (size .ne. 1) then
 64:          if (rank .eq. 0) then
 65:             write(6,*) 'This is a uniprocessor example only!'
 66:          endif

 70: !  Initialize problem parameters
 71:       m = 214
 72:       n = 3


 75: !  Allocate vectors for the solution and gradient
 76:       call VecCreateSeq(PETSC_COMM_SELF,n,x,ierr)
 77:       call VecCreateSeq(PETSC_COMM_SELF,m,f,ierr)


 80: !  The TAO code begins here

 82: !  Create TAO solver
 83:       call TaoCreate(PETSC_COMM_SELF,tao,ierr)
 84:       CHKERRQ(ierr)
 85:       call TaoSetType(tao,TAOPOUNDERS,ierr)
 86:       CHKERRQ(ierr)
 87: !  Set routines for function, gradient, and hessian evaluation

 89:       call TaoSetSeparableObjectiveRoutine(tao,f,                       &
 90:      &      FormFunction,PETSC_NULL_OBJECT,ierr)
 91:       CHKERRQ(ierr)

 93: !  Optional: Set initial guess

 96:       call TaoSetInitialVector(tao, x, ierr)
 97:       CHKERRQ(ierr)


100: !  Check for TAO command line options
101:       call TaoSetFromOptions(tao,ierr)
102:       CHKERRQ(ierr)
103:       call TaoSetHistory(tao,hist,resid,cnorm,                          &
104:      &     100,PETSC_TRUE,ierr)
105:       CHKERRQ(ierr)

107:       call TaoSolve(tao,ierr)

114:       call TaoGetConvergedReason(tao, reason, ierr)

116:          print *,'Try a different TAO method, adjust some parameters,'

118:       endif

138: !  FormFunction - Evaluates the function f(X) and gradient G(X)
139: !
140: !  Input Parameters:
141: !  tao - the Tao context

143: !  dummy - not used

145: !  Output Parameters:
146: !  f - function vector

148:       subroutine FormFunction(tao, x, f, dummy, ierr)
149:       implicit none

151: ! n,alpha defined in chwirut1f.h
152: #include "chwirut1f.h"

154:       Tao        tao
155:       Vec              x,f
156:       PetscErrorCode   ierr
157:       PetscInt         dummy

159:       PetscInt         i

161: ! PETSc's VecGetArray acts differently in Fortran than it does in C.
162: ! Calling VecGetArray((Vec) X, (PetscReal) x_array(0:1), (PetscOffset) x_index, ierr)
163: ! will return an array of doubles referenced by x_array offset by x_index.
164: !  i.e.,  to reference the kth element of X, use x_array(k + x_index).
165: ! Notice that by declaring the arrays with range (0:1), we are using the C 0-indexing practice.
166:       PetscReal        f_v(0:1),x_v(0:1)
167:       PetscOffset      f_i,x_i

169:       0

171: !     Get pointers to vector data
172:       call VecGetArray(x,x_v,x_i,ierr)
173:       CHKERRQ(ierr)
174:       call VecGetArray(f,f_v,f_i,ierr)
175:       CHKERRQ(ierr)


178: !     Compute F(X)
179:       do i=0,m-1
180:          f_v(f_i+i) = y(i) - exp(-x_v(x_i+0)*t(i))/                      &
181:      &    (x_v(x_i+1) + x_v(x_i+2)*t(i))

183:       enddo


186: !     Restore vectors
187:       call VecRestoreArray(X,x_v,x_i,ierr)
188:       CHKERRQ(ierr)
189:       call VecRestoreArray(F,f_v,f_i,ierr)
190:       CHKERRQ(ierr)


193:       return
194:       end
