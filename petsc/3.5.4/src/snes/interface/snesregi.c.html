 36:       This is used by SNESSetType() to make sure that at least one

 38:     DLL then SNESRegisterAll() may be called several times.

 44: /*@C
 45:    SNESRegisterAll - Registers all of the nonlinear solver methods in the SNES package.

 47:    Not Collective

 49:    Level: advanced

 51: .keywords: SNES, register, all

 53: .seealso:  SNESRegisterDestroy()
 54: @*/
 55: PetscErrorCode  SNESRegisterAll(void)
 56: {

 60:   SNESRegisterAllCalled = PETSC_TRUE;

 62:   SNESRegister(SNESNEWTONLS,     SNESCreate_NEWTONLS);
 63:   SNESRegister(SNESNEWTONTR,     SNESCreate_NEWTONTR);
 64:   SNESRegister(SNESTEST,         SNESCreate_Test);
 65:   SNESRegister(SNESNRICHARDSON,  SNESCreate_NRichardson);
 66:   SNESRegister(SNESKSPONLY,      SNESCreate_KSPONLY);
 67:   SNESRegister(SNESVINEWTONRSLS, SNESCreate_VINEWTONRSLS);
 68:   SNESRegister(SNESVINEWTONSSLS, SNESCreate_VINEWTONSSLS);
 69:   SNESRegister(SNESNGMRES,       SNESCreate_NGMRES);
 70:   SNESRegister(SNESQN,           SNESCreate_QN);
 71:   SNESRegister(SNESSHELL,        SNESCreate_Shell);
 72:   SNESRegister(SNESNGS,          SNESCreate_NGS);
 73:   SNESRegister(SNESNCG,          SNESCreate_NCG);
 74:   SNESRegister(SNESFAS,          SNESCreate_FAS);
 75:   SNESRegister(SNESMS,           SNESCreate_MS);
 76:   SNESRegister(SNESNASM,         SNESCreate_NASM);
 77:   SNESRegister(SNESANDERSON,     SNESCreate_Anderson);
 78:   SNESRegister(SNESASPIN,        SNESCreate_ASPIN);
 79:   SNESRegister(SNESCOMPOSITE,    SNESCreate_Composite);
 80:   return(0);
 81: }
