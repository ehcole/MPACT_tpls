398:       if (p[i].a op u[j].a) {                                           \
399:         u[j].a = p[i].a;                                                \
400:         u[j].b = p[i].b;                                                \
401:       } else if (u[j].a == p[i].a) {                                    \
402:         u[j].b = PetscMin(u[j].b,p[i].b);                               \
403:       }                                                                 \
404:       p[i].a = v.a;                                                     \
405:       p[i].b = v.b;                                                     \
406:     }                                                                   \
407:   }
408: #define DEF_PackPair(type1,type2)                                       \
409:   typedef struct {type1 a; type2 b;} PairType(type1,type2);             \
410:   static void CPPJoin3_(Pack_,type1,type2)(PetscInt n,PetscInt bs,const PetscInt *idx,const void *unpacked,void *packed) { \

412:     PairType(type1,type2) *p = (PairType(type1,type2)*)packed;          \

438:     PairType(type1,type2) *u = (PairType(type1,type2)*)unpacked;        \
439:     PairType(type1,type2) *p = (PairType(type1,type2)*)packed;          \
440:     PetscInt i;                                                         \
441:     for (i=0; i<n; i++) {                                               \
442:       PetscInt j = idx[i];                                              \
443:       PairType(type1,type2) v;                                          \
444:       v.a = u[j].a;                                                     \
445:       v.b = u[j].b;                                                     \
446:       u[j].a = p[i].a;                                                  \
447:       u[j].b = p[i].b;                                                  \
448:       p[i].a = v.a;                                                     \
449:       p[i].b = v.b;                                                     \
450:     }                                                                   \
451:   }                                                                     \
452:   static void FetchAndAdd_ ## type1 ## _ ## type2(PetscInt n,PetscInt bs,const PetscInt *idx,void *unpacked,void *packed) { \
453:     PairType(type1,type2) *u = (PairType(type1,type2)*)unpacked;       \
454:     PairType(type1,type2) *p = (PairType(type1,type2)*)packed;         \
455:     PetscInt i;                                                         \
456:     for (i=0; i<n; i++) {                                               \
457:       PetscInt j = idx[i];                                              \
458:       PairType(type1,type2) v;                                          \
459:       v.a = u[j].a;                                                     \
460:       v.b = u[j].b;                                                     \
461:       u[j].a = v.a + p[i].a;                                            \
462:       u[j].b = v.b + p[i].b;                                            \
463:       p[i].a = v.a;                                                     \
464:       p[i].b = v.b;                                                     \
465:     }                                                                   \
466:   }                                                                     \
467:   DEF_UnpackXloc(type1,type2,Max,>)                                     \
468:   DEF_UnpackXloc(type1,type2,Min,<)                                     \

533: static PetscErrorCode PetscSFSetUp_Basic(PetscSF sf)
534: {
535:   PetscSF_Basic *bas = (PetscSF_Basic*)sf->data;
537:   PetscInt *rlengths,*ilengths,i;
538:   MPI_Comm comm;
539:   MPI_Request *rootreqs,*leafreqs;

542:   PetscObjectGetComm((PetscObject)sf,&comm);
543:   PetscObjectGetNewTag((PetscObject)sf,&bas->tag);
544:   /*
545:    * Inform roots about how many leaves and from which ranks
546:    */
547:   PetscMalloc1(sf->nranks,&rlengths);
548:   /* Determine number, sending ranks, and length of incoming  */
549:   for (i=0; i<sf->nranks; i++) {
550:     rlengths[i] = sf->roffset[i+1] - sf->roffset[i]; /* Number of roots referenced by my leaves; for rank sf->ranks[i] */
551:   }
552:   PetscCommBuildTwoSided(comm,1,MPIU_INT,sf->nranks,sf->ranks,rlengths,&bas->niranks,&bas->iranks,(void**)&ilengths);
553:   PetscFree(rlengths);

555:   /* Send leaf identities to roots */
556:   for (i=0,bas->itotal=0; i<bas->niranks; i++) bas->itotal += ilengths[i];
557:   PetscMalloc2(bas->niranks+1,&bas->ioffset,bas->itotal,&bas->irootloc);
558:   PetscMalloc2(bas->niranks,&rootreqs,sf->nranks,&leafreqs);
559:   bas->ioffset[0] = 0;
560:   for (i=0; i<bas->niranks; i++) {
561:     bas->ioffset[i+1] = bas->ioffset[i] + ilengths[i];
562:     MPI_Irecv(bas->irootloc+bas->ioffset[i],ilengths[i],MPIU_INT,bas->iranks[i],bas->tag,comm,&rootreqs[i]);
563:   }
564:   for (i=0; i<sf->nranks; i++) {
565:     PetscMPIInt npoints;
566:     PetscMPIIntCast(sf->roffset[i+1] - sf->roffset[i],&npoints);
567:     MPI_Isend(sf->rremote+sf->roffset[i],npoints,MPIU_INT,sf->ranks[i],bas->tag,comm,&leafreqs[i]);
568:   }
569:   MPI_Waitall(bas->niranks,rootreqs,MPI_STATUSES_IGNORE);
570:   MPI_Waitall(sf->nranks,leafreqs,MPI_STATUSES_IGNORE);
571:   PetscFree(ilengths);
572:   PetscFree2(rootreqs,leafreqs);
573:   return(0);
574: }

578: static PetscErrorCode PetscSFBasicPackTypeSetup(PetscSFBasicPack link,MPI_Datatype unit)
579: {
581:   PetscBool      isInt,isPetscInt,isPetscReal,is2Int,is2PetscInt;
582:   PetscInt       nPetscIntContig,nPetscRealContig;
583: #if defined(PETSC_HAVE_COMPLEX)

585:   PetscInt nPetscComplexContig;
586: #endif

590:   MPIPetsc_Type_compare(unit,MPIU_INT,&isPetscInt);

606: #endif
607:   else if (is2Int) PackInit_int_int(link);
608:   else if (is2PetscInt) PackInit_PetscInt_PetscInt(link);
609:   else if (nPetscIntContig) {
610:     if (nPetscIntContig%7 == 0) PackInit_PetscInt_7(link);
611:     else if (nPetscIntContig%5 == 0) PackInit_PetscInt_5(link);
612:     else if (nPetscIntContig%4 == 0) PackInit_PetscInt_4(link);
613:     else if (nPetscIntContig%3 == 0) PackInit_PetscInt_3(link);
614:     else if (nPetscIntContig%2 == 0) PackInit_PetscInt_2(link);
615:     else PackInit_PetscInt(link);
616:     link->bs = nPetscIntContig;
617:     link->unitbytes *= nPetscIntContig;
618:   } else if (nPetscRealContig) {
619:     if (nPetscRealContig%7 == 0) PackInit_PetscReal_7(link);
620:     else if (nPetscRealContig%5 == 0) PackInit_PetscReal_5(link);
621:     else if (nPetscRealContig%4 == 0) PackInit_PetscReal_4(link);
622:     else if (nPetscRealContig%3 == 0) PackInit_PetscReal_3(link);
623:     else if (nPetscRealContig%2 == 0) PackInit_PetscReal_2(link);
624:     else PackInit_PetscReal(link);
625:     link->bs = nPetscRealContig;
626:     link->unitbytes *= nPetscRealContig;
627: #if defined(PETSC_HAVE_COMPLEX)
628:   } else if (nPetscComplexContig) {
629:     if (nPetscComplexContig%7 == 0) PackInit_PetscComplex_7(link);
630:     else if (nPetscComplexContig%5 == 0) PackInit_PetscComplex_5(link);
631:     else if (nPetscComplexContig%4 == 0) PackInit_PetscComplex_4(link);
632:     else if (nPetscComplexContig%3 == 0) PackInit_PetscComplex_3(link);
633:     else if (nPetscComplexContig%2 == 0) PackInit_PetscComplex_2(link);
634:     else PackInit_PetscComplex_1(link);
635:     link->bs = nPetscComplexContig;
636:     link->unitbytes *= nPetscComplexContig;
637: #endif
638:   } else {
639:     PetscMPIInt bytes;
640:     MPI_Type_size(unit,&bytes);
641:     if (bytes % sizeof(int)) SETERRQ1(PETSC_COMM_SELF,PETSC_ERR_SUP,"No support for type size not divisible by %D",sizeof(int));
642:     switch (bytes / sizeof(int)) {
643:     case 1: PackInit_block_int_1(link); break;
644:     case 2: PackInit_block_int_2(link); break;
645:     case 3: PackInit_block_int_3(link); break;
646:     case 4: PackInit_block_int_4(link); break;
647:     case 5: PackInit_block_int_5(link); break;
648:     case 6: PackInit_block_int_6(link); break;
649:     case 7: PackInit_block_int_7(link); break;
650:     case 8: PackInit_block_int_8(link); break;
651:     default: SETERRQ(PETSC_COMM_SELF,PETSC_ERR_SUP,"No support for arbitrary block sizes");
652:     }
653:   }
654:   MPI_Type_dup(unit,&link->unit);
655:   return(0);
656: }

755: static PetscErrorCode PetscSFBasicGetPack(PetscSF sf,MPI_Datatype unit,const void *key,PetscSFBasicPack *mylink)
756: {
757:   PetscSF_Basic    *bas = (PetscSF_Basic*)sf->data;
758:   PetscErrorCode   ierr;
759:   PetscSFBasicPack link,*p;
760:   PetscInt         nrootranks,nleafranks;
761:   const PetscInt   *rootoffset,*leafoffset;

764:   /* Look for types in cache */
765:   for (p=&bas->avail; (link=*p); p=&link->next) {
766:     PetscBool match;
767:     MPIPetsc_Type_compare(unit,link->unit,&match);
768:     if (match) {
769:       *p = link->next;          /* Remove from available list */
770:       goto found;
771:     }
772:   }

774:   /* Create new composite types for each send rank */
775:   PetscSFBasicGetRootInfo(sf,&nrootranks,NULL,&rootoffset,NULL);
776:   PetscSFBasicGetLeafInfo(sf,&nleafranks,NULL,&leafoffset,NULL);
777:   PetscNew(&link);
778:   PetscSFBasicPackTypeSetup(link,unit);
779:   PetscMalloc2(rootoffset[nrootranks]*link->unitbytes,&link->root,leafoffset[nleafranks]*link->unitbytes,&link->leaf);
780:   PetscMalloc1((nrootranks+nleafranks),&link->requests);

782: found:
783:   link->key  = key;
784:   link->next = bas->inuse;
785:   bas->inuse = link;

787:   *mylink = link;
788:   return(0);
789: }

899: /* Send from roots to leaves */
900: static PetscErrorCode PetscSFBcastBegin_Basic(PetscSF sf,MPI_Datatype unit,const void *rootdata,void *leafdata)
901: {
902:   PetscSF_Basic     *bas = (PetscSF_Basic*)sf->data;
903:   PetscErrorCode    ierr;
904:   PetscSFBasicPack  link;
905:   PetscInt          i,nrootranks,nleafranks;
906:   const PetscInt    *rootoffset,*leafoffset,*rootloc,*leafloc;
907:   const PetscMPIInt *rootranks,*leafranks;
908:   MPI_Request       *rootreqs,*leafreqs;
909:   size_t            unitbytes;

912:   PetscSFBasicGetRootInfo(sf,&nrootranks,&rootranks,&rootoffset,&rootloc);
913:   PetscSFBasicGetLeafInfo(sf,&nleafranks,&leafranks,&leafoffset,&leafloc);
914:   PetscSFBasicGetPack(sf,unit,rootdata,&link);

916:   unitbytes = link->unitbytes;

918:   PetscSFBasicPackGetReqs(sf,link,&rootreqs,&leafreqs);
919:   /* Eagerly post leaf receives */
920:   for (i=0; i<nleafranks; i++) {
921:     PetscMPIInt n = leafoffset[i+1] - leafoffset[i];
922:     MPI_Irecv(link->leaf+leafoffset[i]*unitbytes,n,unit,leafranks[i],bas->tag,PetscObjectComm((PetscObject)sf),&leafreqs[i]);
923:   }
924:   /* Pack and send root data */
925:   for (i=0; i<nrootranks; i++) {
926:     PetscMPIInt n          = rootoffset[i+1] - rootoffset[i];
927:     void        *packstart = link->root+rootoffset[i]*unitbytes;
928:     (*link->Pack)(n,link->bs,rootloc+rootoffset[i],rootdata,packstart);
929:     MPI_Isend(packstart,n,unit,rootranks[i],bas->tag,PetscObjectComm((PetscObject)sf),&rootreqs[i]);
930:   }
931:   return(0);
932: }

936: PetscErrorCode PetscSFBcastEnd_Basic(PetscSF sf,MPI_Datatype unit,const void *rootdata,void *leafdata)
937: {
938:   PetscErrorCode   ierr;
939:   PetscSFBasicPack link;
940:   PetscInt         i,nleafranks;
941:   const PetscInt   *leafoffset,*leafloc;

944:   PetscSFBasicGetPackInUse(sf,unit,rootdata,PETSC_OWN_POINTER,&link);
945:   PetscSFBasicPackWaitall(sf,link);
946:   PetscSFBasicGetLeafInfo(sf,&nleafranks,NULL,&leafoffset,&leafloc);
947:   for (i=0; i<nleafranks; i++) {
948:     PetscMPIInt n          = leafoffset[i+1] - leafoffset[i];
949:     const void  *packstart = link->leaf+leafoffset[i]*link->unitbytes;
950:     (*link->UnpackInsert)(n,link->bs,leafloc+leafoffset[i],leafdata,packstart);
951:   }
952:   PetscSFBasicReclaimPack(sf,&link);
953:   return(0);
954: }

1033: {

1035:   void              (*FetchAndOp)(PetscInt,PetscInt,const PetscInt*,void*,void*);
1036:   PetscErrorCode    ierr;

1038:   PetscInt          i,nrootranks,nleafranks;
1039:   const PetscInt    *rootoffset,*leafoffset,*rootloc,*leafloc;
1040:   const PetscMPIInt *rootranks,*leafranks;
1041:   MPI_Request       *rootreqs,*leafreqs;
1042:   size_t            unitbytes;

1045:   PetscSFBasicGetPackInUse(sf,unit,rootdata,PETSC_OWN_POINTER,&link);
1046:   /* This implementation could be changed to unpack as receives arrive, at the cost of non-determinism */

1048:   unitbytes = link->unitbytes;
1049:   PetscSFBasicGetRootInfo(sf,&nrootranks,&rootranks,&rootoffset,&rootloc);
1050:   PetscSFBasicGetLeafInfo(sf,&nleafranks,&leafranks,&leafoffset,&leafloc);
1051:   PetscSFBasicPackGetReqs(sf,link,&rootreqs,&leafreqs);
1052:   /* Post leaf receives */
1053:   for (i=0; i<nleafranks; i++) {
