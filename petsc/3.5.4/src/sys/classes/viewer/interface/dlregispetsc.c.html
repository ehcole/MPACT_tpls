 46:   if (PetscSysPackageInitialized) return(0);
 47:   PetscSysPackageInitialized = PETSC_TRUE;
 48:   /* Register Classes */

 50:   PetscClassIdRegister("Container",&PETSC_CONTAINER_CLASSID);

 52:   /* Register Events */

 54:   /* Process info exclusions */
