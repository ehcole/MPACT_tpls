425: PetscErrorCode MatScale_MPIDense(Mat inA,PetscScalar alpha)
426: {
427:   Mat_MPIDense   *A     = (Mat_MPIDense*)inA->data;
428:   Mat_SeqDense   *a     = (Mat_SeqDense*)A->A->data;
429:   PetscScalar    oalpha = alpha;
