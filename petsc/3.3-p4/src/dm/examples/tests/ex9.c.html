  2: static char help[] = "Tests DMCreateColoring() in 3d.\n\n";

  4: #include <petscmat.h>
  5: #include <petscdmda.h>

  9: int main(int argc,char **argv)
 10: {
 11:   PetscInt       i,M = 3,N = 5,P=3,s=1,w=2,m = PETSC_DECIDE,n = PETSC_DECIDE,p = PETSC_DECIDE;
 13:   PetscInt       *lx = PETSC_NULL,*ly = PETSC_NULL,*lz = PETSC_NULL;
 14:   DM             da;
 15:   PetscBool      flg = PETSC_FALSE,test_order = PETSC_FALSE;
 16:   ISColoring     coloring;
 17:   Mat            mat;
 18:   DMDAStencilType  stencil_type = DMDA_STENCIL_BOX;
 19:   Vec            lvec,dvec;
 20:   MatFDColoring  fdcoloring;

 22:   PetscInitialize(&argc,&argv,(char*)0,help);

 24:   /* Read options */
 25:   PetscOptionsGetInt(PETSC_NULL,"-M",&M,PETSC_NULL);
 26:   PetscOptionsGetInt(PETSC_NULL,"-N",&N,PETSC_NULL);
 27:   PetscOptionsGetInt(PETSC_NULL,"-P",&P,PETSC_NULL);
 28:   PetscOptionsGetInt(PETSC_NULL,"-m",&m,PETSC_NULL);
 29:   PetscOptionsGetInt(PETSC_NULL,"-n",&n,PETSC_NULL);
 30:   PetscOptionsGetInt(PETSC_NULL,"-p",&p,PETSC_NULL);
 31:   PetscOptionsGetInt(PETSC_NULL,"-s",&s,PETSC_NULL);
 32:   PetscOptionsGetInt(PETSC_NULL,"-w",&w,PETSC_NULL);
 33:   PetscOptionsGetBool(PETSC_NULL,"-star",&flg,PETSC_NULL);
 34:   if (flg) stencil_type =  DMDA_STENCIL_STAR;
 35:   PetscOptionsGetBool(PETSC_NULL,"-test_order",&test_order,PETSC_NULL);
 36:   flg  = PETSC_FALSE;
 37:   PetscOptionsGetBool(PETSC_NULL,"-distribute",&flg,PETSC_NULL);
 38:   if (flg) {
 39:     if (m == PETSC_DECIDE) SETERRQ(PETSC_COMM_WORLD,1,"Must set -m option with -distribute option");
 40:     PetscMalloc(m*sizeof(PetscInt),&lx);
 41:     for (i=0; i<m-1; i++) { lx[i] = 4;}
 42:     lx[m-1] = M - 4*(m-1);
 43:     if (n == PETSC_DECIDE) SETERRQ(PETSC_COMM_WORLD,1,"Must set -n option with -distribute option");
 44:     PetscMalloc(n*sizeof(PetscInt),&ly);
 45:     for (i=0; i<n-1; i++) { ly[i] = 2;}
 46:     ly[n-1] = N - 2*(n-1);
 47:     if (p == PETSC_DECIDE) SETERRQ(PETSC_COMM_WORLD,1,"Must set -p option with -distribute option");
 48:     PetscMalloc(p*sizeof(PetscInt),&lz);
 49:     for (i=0; i<p-1; i++) { lz[i] = 2;}
 50:     lz[p-1] = P - 2*(p-1);
 51:   }

 53:   /* Create distributed array and get vectors */
 54:   DMDACreate3d(PETSC_COMM_WORLD,DMDA_BOUNDARY_NONE,DMDA_BOUNDARY_NONE,DMDA_BOUNDARY_NONE,stencil_type,M,N,P,m,n,p,w,s,lx,ly,lz,&da);
 55:   PetscFree(lx);
 56:   PetscFree(ly);
 57:   PetscFree(lz);

 59:   DMCreateColoring(da,IS_COLORING_GLOBAL,MATMPIAIJ,&coloring);
 60:   DMCreateMatrix(da,MATMPIAIJ,&mat);
 61:   MatFDColoringCreate(mat,coloring,&fdcoloring);

 63:   DMCreateGlobalVector(da,&dvec);
 64:   DMCreateLocalVector(da,&lvec);

 66:   /* Free memory */
 67:   MatFDColoringDestroy(&fdcoloring);
 68:   VecDestroy(&dvec);
 69:   VecDestroy(&lvec);
 70:   MatDestroy(&mat);
 71:   ISColoringDestroy(&coloring);
 72:   DMDestroy(&da);
 73:   PetscFinalize();
 74:   return 0;
 75: }
