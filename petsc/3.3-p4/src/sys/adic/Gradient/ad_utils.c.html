 13: void PetscADSetValueAndColor(DERIV_TYPE *vars,int n,ISColoringValue *colors,double *values)
 14: {
 15:   int       i,j;
 16:   PetscReal *d;

 18:   for (i=0; i<n; i++) {
 19:     DERIV_val(vars[i]) = values[i];
 20:     d = (PetscReal*)DERIV_grad(vars[i]);
 21:     for (j=0; j<ad_GRAD_MAX; j++) {
 22:       d[j] = 0.0;
 23:     }
 24:     d[colors[i]] = 1.0;
 25:   }
 26: }
