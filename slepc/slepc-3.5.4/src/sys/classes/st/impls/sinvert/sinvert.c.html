 28: PetscErrorCode STApply_Sinvert(ST st,Vec x,Vec y)
 29: {

 33:   if (st->nmat>1) {
 34:     /* generalized eigenproblem: y = (A - sB)^-1 B x */
 35:     MatMult(st->T[0],x,st->w);
 36:     STMatSolve(st,st->w,y);
 37:   } else {
 38:     /* standard eigenproblem: y = (A - sI)^-1 x */
 39:     STMatSolve(st,x,y);
 40:   }
 41:   return(0);
 42: }

 46: PetscErrorCode STApplyTranspose_Sinvert(ST st,Vec x,Vec y)
 47: {

 51:   if (st->nmat>1) {
 52:     /* generalized eigenproblem: y = B^T (A - sB)^-T x */
 53:     STMatSolveTranspose(st,x,st->w);
 54:     MatMultTranspose(st->T[0],st->w,y);
 55:   } else {
 56:     /* standard eigenproblem: y = (A - sI)^-T x */
 57:     STMatSolveTranspose(st,x,y);
 58:   }
 59:   return(0);
 60: }

 64: PetscErrorCode STBackTransform_Sinvert(ST st,PetscInt n,PetscScalar *eigr,PetscScalar *eigi)
 65: {
 66:   PetscInt    j;
 67: #if !defined(PETSC_USE_COMPLEX)
 68:   PetscScalar t;
 69: #endif

 72: #if !defined(PETSC_USE_COMPLEX)
 73:   for (j=0;j<n;j++) {
 74:     if (eigi[j] == 0) eigr[j] = 1.0 / eigr[j] + st->sigma;
 75:     else {
 76:       t = eigr[j] * eigr[j] + eigi[j] * eigi[j];
 77:       eigr[j] = eigr[j] / t + st->sigma;
 78:       eigi[j] = - eigi[j] / t;
 79:     }
 80:   }
 81: #else
 82:   for (j=0;j<n;j++) {
 83:     eigr[j] = 1.0 / eigr[j] + st->sigma;
 84:   }
 85: #endif
 86:   return(0);
 87: }

 91: PetscErrorCode STPostSolve_Sinvert(ST st)
 92: {

 96:   if (st->shift_matrix == ST_MATMODE_INPLACE) {
 97:     if (st->nmat>1) {
 98:       MatAXPY(st->A[0],st->sigma,st->A[1],st->str);
 99:     } else {
100:       MatShift(st->A[0],st->sigma);
101:     }
102:     st->Astate[0] = ((PetscObject)st->A[0])->state;
103:     st->setupcalled = 0;
104:   }
105:   return(0);
106: }

110: PetscErrorCode STSetUp_Sinvert(ST st)
111: {
113:   PetscInt       k,nc,nmat=PetscMax(st->nmat,2);
114:   PetscScalar    *coeffs=NULL;

117:   /* if the user did not set the shift, use the target value */
118:   if (!st->sigma_set) st->sigma = st->defsigma;
119:   if (nmat<3 || st->transform) {
120:     if (nmat>2) {
121:       nc = (nmat*(nmat+1))/2;
122:       PetscMalloc(nc*sizeof(PetscScalar),&coeffs);
123:       /* Compute coeffs */
124:       STCoeffs_Monomial(st,coeffs);
125:     }
126:     /* T[0] = A_n */
127:     k = nmat-1;
128:     PetscObjectReference((PetscObject)st->A[k]);
129:     st->T[0] = st->A[k];
130:     for (k=1;k<nmat;k++) {
131:       STMatMAXPY_Private(st,nmat>2?st->sigma:-st->sigma,0.0,nmat-k-1,coeffs?coeffs+(k*(k+1))/2:NULL,PETSC_TRUE,&st->T[k]);
132:     }
133:     if (nmat>2) { PetscFree(coeffs); }
134:     st->P = st->T[nmat-1];
135:     PetscObjectReference((PetscObject)st->P);
136:   } else {
137:     for (k=0;k<nmat;k++) {
138:       PetscObjectReference((PetscObject)st->A[k]);
139:       st->T[k] = st->A[k];
140:     }
141:   } 
142:   if (st->P) {
143:     if (!st->ksp) { STGetKSP(st,&st->ksp); }
144:     KSPSetOperators(st->ksp,st->P,st->P);
145:     KSPSetUp(st->ksp);
146:   }
147:   return(0);
148: }

152: PetscErrorCode STSetShift_Sinvert(ST st,PetscScalar newshift)
153: {
155:   PetscInt       nmat=PetscMax(st->nmat,2),k,nc;
156:   PetscScalar    *coeffs=NULL;

159:   /* Nothing to be done if STSetUp has not been called yet */
160:   if (!st->setupcalled) return(0);
161:   if (nmat<3 || st->transform) {
162:     if (st->shift_matrix == ST_MATMODE_COPY && nmat>2) {
163:       nc = (nmat*(nmat+1))/2;
164:       PetscMalloc(nc*sizeof(PetscScalar),&coeffs);
165:       /* Compute coeffs */
166:       STCoeffs_Monomial(st,coeffs);
167:     }
168:     for (k=1;k<nmat;k++) {
169:       STMatMAXPY_Private(st,nmat>2?newshift:-newshift,nmat>2?st->sigma:-st->sigma,nmat-k-1,coeffs?coeffs+(k*(k+1))/2:NULL,PETSC_FALSE,&st->T[k]);
170:     }
171:     if (st->shift_matrix == ST_MATMODE_COPY && nmat>2) {
172:       PetscFree(coeffs);
173:     }
174:     if (st->P!=st->T[nmat-1]) {
175:       MatDestroy(&st->P);
176:       st->P = st->T[nmat-1];
177:       PetscObjectReference((PetscObject)st->P);
178:     }
179:   }
180:   if (st->P) {
181:     if (!st->ksp) { STGetKSP(st,&st->ksp); }
182:     KSPSetOperators(st->ksp,st->P,st->P);
183:     KSPSetUp(st->ksp);
184:   }
185:   return(0);
186: }

190: PetscErrorCode STSetFromOptions_Sinvert(ST st)
191: {
193:   PC             pc;
194:   PCType         pctype;
195:   KSPType        ksptype;

198:   if (!st->ksp) { STGetKSP(st,&st->ksp); }
199:   KSPGetPC(st->ksp,&pc);
200:   KSPGetType(st->ksp,&ksptype);
201:   PCGetType(pc,&pctype);
202:   if (!pctype && !ksptype) {
203:     if (st->shift_matrix == ST_MATMODE_SHELL) {
204:       /* in shell mode use GMRES with Jacobi as the default */
205:       KSPSetType(st->ksp,KSPGMRES);
206:       PCSetType(pc,PCJACOBI);
207:     } else {
208:       /* use direct solver as default */
209:       KSPSetType(st->ksp,KSPPREONLY);
210:       PCSetType(pc,PCREDUNDANT);
211:     }
212:   }
213:   return(0);
214: }
